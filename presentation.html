CREATE TRIGGER after_booking_completed
AFTER UPDATE ON Bookings
FOR EACH ROW
BEGIN
    IF NEW.status = 'Completed' THEN
        UPDATE Lockers 
        SET status = 'Available'
        WHERE locker_id = NEW.locker_id;
    END IF;
END
